<template>
  <panel-documentos :anexo-upload="anexoUpload"
                    :doc-novo="docNovo"
                    :documentos="documentos"
                    :tipos-documento="tiposDocumento"
                    @criarNovoDocumento="tratarEventoCriarNovoDocumento"
                    @buscarTodosDocumentos="tratarEventoBuscarTodosDocumentos"
                    @buscarTipoDocumentos="tratarEventoBuscarTipoDocumentos"
                    @salvarDocumento="tratarEventoSalvarDocumento"
                    @editarDocumento="tratarEventoEditarDocumento"
                    @uploadAnexo="tratarEventoUploadAnexo"
                    @removerDocumento="tratarEventoRemoverDocumento"
                    @resetarDocumentoNovo="tratarEventoResetarDocumentoNovo"/>
</template>
 
<script>
    import { mapActions } from 'vuex'
    import { actionTypes } from '@/core/constants'
    import PanelDocumentos from '@/views/components/panel/panel-documentos/PanelDocumentos'
 
    export default {
        name: 'VisualizarRegistroMovimentacaoInternaDocumentos',
        components: { PanelDocumentos },
        props: {
            movimentacaoInternaId: {
                type: Number,
                required: true
            }
        },
        data () {
            return {
                documentos: [],
                tiposDocumento: [],
                docNovo: {},
                anexoUpload: '',
            }
        },
        watch: {
            async movimentacaoInternaId() {
                this.docNovo = {}
                this.anexoUpload = ''
                this.documentos = await this.buscarDocumentosMovimentacaoInterna(this.movimentacaoInternaId)
            }
        },
        methods: {
            ...mapActions([
                actionTypes.MOVIMENTACAO_INTERNA.DOCUMENTO.BUSCAR_DOCUMENTOS_MOVIMENTACAO_INTERNA,
                actionTypes.MOVIMENTACAO_INTERNA.DOCUMENTO.BUSCAR_TIPO_DOCUMENTO_MOVIMENTACAO_INTERNA,
                actionTypes.MOVIMENTACAO_INTERNA.DOCUMENTO.CADASTRAR_DOCUMENTO_MOVIMENTACAO_INTERNA,
                actionTypes.MOVIMENTACAO_INTERNA.DOCUMENTO.ATUALIZAR_DOCUMENTO_MOVIMENTACAO_INTERNA,
                actionTypes.MOVIMENTACAO_INTERNA.DOCUMENTO.DELETAR_DOCUMENTO_MOVIMENTACAO_INTERNA,
                actionTypes.PATRIMONIO.INCORPORACAO.DOCUMENTO.DOCUMENTO_UPLOAD,
            ]),
            tratarEventoCriarNovoDocumento () {
                this.docNovo = { movimentacao: this.movimentacaoInternaId }
            },
            async tratarEventoBuscarTodosDocumentos () {
                this.documentos = await this.buscarDocumentosMovimentacaoInterna(this.movimentacaoInternaId)
            },
            async tratarEventoBuscarTipoDocumentos () {
                this.tiposDocumento = await this.buscarTipoDocumentoMovimentacaoInterna()
            },
            async tratarEventoSalvarDocumento (documento) {
                await this.cadastrarDocumentoMovimentacaoInterna(documento)
                await this.tratarEventoBuscarTodosDocumentos()
            },
            async tratarEventoEditarDocumento (documento) {
                await this.atualizarDocumentoMovimentacaoInterna(documento)
            },
            async tratarEventoUploadAnexo (anexo) {
                this.anexoUpload = await this.documentoUpload(anexo)
            },
            async tratarEventoRemoverDocumento (documento) {
                await this.deletarDocumentoMovimentacaoInterna(documento)
            },
            tratarEventoResetarDocumentoNovo () {
                this.docNovo = {}
            },
        },
    }
</script>
